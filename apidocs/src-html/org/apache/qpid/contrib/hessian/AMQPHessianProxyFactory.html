043    /**
044     * Factory for creating Hessian client stubs. The returned stub will
045     * call the remote object for all methods.
046     *
047     * <pre>
048     * HelloHome hello = (HelloHome) factory.create(HelloHome.class, null);
049     * </pre>
050     *
051     * After creation, the stub can be like a regular Java class. Because
052     * it makes remote calls, it can throw more exceptions than a Java class.
053     * In particular, it may throw protocol exceptions.
054     * 
055     * This class is derived from {@link com.caucho.hessian.client.HessianProxyFactory}. 
056     * 
057     * @author Emmanuel Bourg
058     * @author Scott Ferguson
059     */
060    public class AMQPHessianProxyFactory implements ServiceProxyFactory
061    {
062        private SerializerFactory _serializerFactory;
063        private HessianRemoteResolver _resolver;

076        private boolean isHessian2Request = true;

085        /**
086         * Creates the new proxy factory.
087         */
088        public AMQPHessianProxyFactory()
089        {
090            _resolver = new HessianRemoteResolver()
091            {
092                public Object lookup(String type, String url) throws IOException
093                {
094                    ClassLoader loader = Thread.currentThread().getContextClassLoader();
095    
096                    try
097                    {
098                        Class api = Class.forName(type, false, loader);
099                        return create(api, url);
100                    }
101                    catch (Exception e)
102                    {
103                        throw new IOException(String.valueOf(e));
104                    }
105                }
106            };
107        }

282        /**
283         * Creates a new proxy with the specified URL.  The returned object
284         * is a proxy with the interface specified by api.
285         *
286         * <pre>
287         * String url = "amqp://user:password@localhost:5672/vhost/queue");
288         * HelloHome hello = (HelloHome) factory.create(HelloHome.class, url);
289         * </pre>
290         *
291         * @param api the interface the proxy class needs to implement
292         * @param urlName the URL where the client object is located.
293         * @return a proxy to the object with the specified interface.
294         */

296        {
297            return create(api, urlName, api.getClassLoader());
298        }

300        /**
301         * Creates a new proxy with the specified URL.  The returned object
302         * is a proxy with the interface specified by api.
303         *
304         * <pre>
305         * String url = "amqp://user:password@localhost:5672/vhost/queue");
306         * HelloHome hello = (HelloHome) factory.create(HelloHome.class, url);
307         * </pre>
308         *
309         * @param api the interface the proxy class needs to implement
310         * @param urlName the URL where the client object is located.
311         * @return a proxy to the object with the specified interface.
312         */
313        public Object create(Class api, String urlName, ClassLoader loader) throws MalformedURLException
314        {
315            try
316            {
317                URI uri = new URI(urlName);
318    
319                hostname = uri.getHost();
320                if (uri.getPort() != -1)
321                {
322                    port = uri.getPort();
323                }
324    
325                String userinfo = uri.getUserInfo();
326                if (userinfo != null)
327                {
328                    String[] parts = userinfo.split(":");
329                    user = parts[0];
330                    if (parts.length > 0)
331                    {
332                        password = parts[1];
333                    }
334                }
335    
336                Pattern pattern = Pattern.compile("/([^/]+)(/(.*))?");
337                Matcher matcher = pattern.matcher(uri.getPath());
338                if (matcher.matches())
339                {
340                    virtualhost = matcher.group(1);
341                    if (matcher.groupCount() > 1)
342                    {
343                        queuePrefix = matcher.group(3);
344                    }
345                }
346            }
347            catch (URISyntaxException e)
348            {
349                throw (MalformedURLException) new MalformedURLException().initCause(e);
350            }
351            
352            AMQPHessianProxy handler = new AMQPHessianProxy(this);
353    
354            return Proxy.newProxyInstance(loader, new Class[]{api}, handler);
355        }

362        AbstractHessianInput getHessian1Input(InputStream is)
363        {
364            AbstractHessianInput in;
365    
366            if (debug)
367            {
368                is = new HessianDebugInputStream(is, new PrintWriter(System.out));
369            }
370    
371            in = new HessianInput(is);
372    
373            in.setRemoteResolver(getRemoteResolver());
374    
375            in.setSerializerFactory(getSerializerFactory());
376    
377            return in;
378        }
379    
380        AbstractHessianInput getHessian2Input(InputStream is)
381        {
382            AbstractHessianInput in;
383    
384            if (debug)
385            {
386                is = new HessianDebugInputStream(is, new PrintWriter(System.out));
387            }
388    
389            in = new Hessian2Input(is);
390    
391            in.setRemoteResolver(getRemoteResolver());
392    
393            in.setSerializerFactory(getSerializerFactory());
394    
395            return in;
396        }
397    
398        AbstractHessianOutput getHessianOutput(OutputStream os)
399        {
400            AbstractHessianOutput out;
401    
402            if (isHessian2Request)
403            {
404                out = new Hessian2Output(os);
405            }
406            else
407            {
408                HessianOutput out1 = new HessianOutput(os);
409                out = out1;
410    
411                if (isHessian2Reply)
412                {
413                    out1.setVersion(2);
414                }
415            }
416    
417            out.setSerializerFactory(getSerializerFactory());
418    
419            return out;
420        }
